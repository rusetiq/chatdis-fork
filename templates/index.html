Embed on Your Website
Add ChatDIS to any page with a single script tag. The widget appears as a floating chat button in the corner.
<!-- ChatDIS Widget -->
<script>
  window.ChatDISConfig = {
    serverUrl: 'https://your-chatdis-server.com',
    apiKey: 'your-secret-key',
    position: 'bottom-right'
  };
</script>
<script src="https://your-chatdis-server.com/static/chatdis-loader.js"></script>
